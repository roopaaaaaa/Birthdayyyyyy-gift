  <script>
    function startConfetti() {
      const duration = 4 * 1000;
      const end = Date.now() + duration;

      (function frame() {
        confetti({
          particleCount: 6,
          angle: 60,
          spread: 60,
          origin: { x: 0 }
        });

        confetti({
          particleCount: 6,
          angle: 120,
          spread: 60,
          origin: { x: 1 }
        });

        if (Date.now() < end) {
          requestAnimationFrame(frame);
        }
      })();
    }

    // Auto confetti on load
    window.onload = () => {
      setTimeout(startConfetti, 800);
    };
  </script>
